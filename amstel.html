function updateFeatureSettingsValue() {
var featureSettingsString = "";
sliders = document.querySelectorAll("input");
for (var i = 0; i < sliders.length; ++i) {
featureSettingsString += "\"" + sliders[i].id + "\" " + sliders[i].value / 1000;
if (i < sliders.length - 1)
featureSettingsString += ",";
}
// console.log(featureSettingsString);
document.styleSheets[0].cssRules[2].style.fontVariationSettings = featureSettingsString;
}
window.onload = () => {
sliders = document.querySelectorAll("input");
for (var slider of sliders) {
slider.oninput = updateFeatureSettingsValue;
slider.onchange = updateFeatureSettingsValue;
}
}
</script>
